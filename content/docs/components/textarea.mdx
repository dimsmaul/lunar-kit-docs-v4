# Textarea

A multi-line text input field for longer content.

## Installation

```bash
lunar add textarea
```

## Usage

```tsx
import { Textarea } from '@/components/ui/textarea';
```

### Basic Usage

```tsx
const [value, setValue] = useState('');

<Textarea
  value={value}
  onChangeText={setValue}
  placeholder="Enter your message..."
/>
```

### With Label

```tsx
<Textarea
  label="Description"
  value={description}
  onChangeText={setDescription}
  placeholder="Enter a description..."
/>
```

### With Error

```tsx
<Textarea
  label="Bio"
  value={bio}
  onChangeText={setBio}
  error="Bio is required"
/>
```

### Custom Rows

```tsx
// Small (2 rows)
<Textarea rows={2} placeholder="Short input..." />

// Medium (4 rows - default)
<Textarea rows={4} placeholder="Default size..." />

// Large (8 rows)
<Textarea rows={8} placeholder="Larger input..." />
```

### Disabled State

```tsx
<Textarea
  label="Notes"
  value="This cannot be edited"
  editable={false}
/>
```

### Character Count

```tsx
export function TextareaWithCount({ maxLength = 280 }) {
  const [value, setValue] = useState('');

  return (
    <View className="gap-1">
      <Textarea
        value={value}
        onChangeText={setValue}
        placeholder="What's on your mind?"
        maxLength={maxLength}
      />
      <Text
        variant="muted"
        size="sm"
        align="right"
        className={value.length >= maxLength ? 'text-destructive' : ''}
      >
        {value.length}/{maxLength}
      </Text>
    </View>
  );
}
```

## API Reference

### Textarea

| Prop | Type | Default | Description |
|------|------|---------|-------------|
| `label` | `string` | — | Input label |
| `error` | `string` | — | Error message |
| `rows` | `number` | `4` | Number of visible lines |
| `containerClassName` | `string` | — | Classes for container |
| `className` | `string` | — | Classes for textarea |

All standard `TextInput` props are also supported.

### Review Form

```tsx
export function ReviewForm({ onSubmit }) {

  const [review, setReview] = useState('');

      <CardHeader>
```

### Form Integration

```tsx
import { Controller, useForm } from 'react-hook-form';

export function FormWithTextarea() {
  const { control, handleSubmit } = useForm();

  return (
    <Controller
      control={control}
      name="description"
      rules={{
        required: 'Description is required',
        minLength: {
          value: 20,
          message: 'Description must be at least 20 characters',
        },
      }}
      render={({ field: { value, onChange }, fieldState: { error } }) => (
        <Textarea
          label="Description"
          value={value}
          onChangeText={onChange}
          placeholder="Describe your item..."
          error={error?.message}
          rows={5}
        />
      )}
    />
  );
}
```

### Feedback Form

```tsx
export function FeedbackForm() {
  const [feedback, setFeedback] = useState('');
  const [submitted, setSubmitted] = useState(false);

  const handleSubmit = () => {
    // Submit feedback
    setSubmitted(true);
  };

  if (submitted) {
    return (
      <Banner variant="success" title="Thank you!">
        Your feedback has been submitted.
      </Banner>
    );
  }

  return (
    <View className="gap-4">
      <Text variant="title">We'd love your feedback</Text>
      <Text variant="muted">
        Help us improve by sharing your thoughts.
      </Text>
      <Textarea
        value={feedback}
        onChangeText={setFeedback}
        placeholder="What can we do better?"
        rows={6}
      />
      <Button
        disabled={!feedback.trim()}
        onPress={handleSubmit}
      >
        Submit Feedback
      </Button>
    </View>
  );
}
```

### Auto-expanding Textarea

```tsx
export function AutoExpandingTextarea({ value, onChangeText, ...props }) {
  const [height, setHeight] = useState(100);

  return (
    <TextInput
      value={value}
      onChangeText={onChangeText}
      multiline
      onContentSizeChange={(event) => {
        setHeight(Math.max(100, event.nativeEvent.contentSize.height));
      }}
      style={{ minHeight: height }}
      className="border border-input rounded-lg p-3 text-foreground"
      {...props}
    />
  );
}
```
