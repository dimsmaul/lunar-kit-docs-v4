# Select

A dropdown component for selecting from a list of options.

## Installation

```bash
lunar add select
# For bottom sheet select
lunar add select-sheet
```

## Usage

### Select (Dialog based)

```tsx
import {
  Select,
  SelectTrigger,
  SelectValue,
  SelectContent,
  SelectItem,
  SelectGroup,
  SelectLabel,
  SelectSeparator,
} from '@/components/ui/select';

const [value, setValue] = useState('');

<Select value={value} onValueChange={setValue}>
  <SelectTrigger>
    <SelectValue placeholder="Select an option" />
  </SelectTrigger>
  <SelectContent>
    <SelectItem value="option1" label="Option 1" />
    <SelectItem value="option2" label="Option 2" />
    <SelectItem value="option3" label="Option 3" />
  </SelectContent>
</Select>
```

### SelectSheet (Bottom sheet based)

```tsx
import { SelectSheet, MultiSelectSheet } from '@/components/ui/select-sheet';

const options = [
  { label: 'Apple', value: 'apple' },
  { label: 'Banana', value: 'banana' },
  { label: 'Orange', value: 'orange' },
];

// Single select
<SelectSheet
  options={options}
  value={value}
  onValueChange={setValue}
  placeholder="Select a fruit"
  title="Choose a Fruit"
/>

// Multi select
<MultiSelectSheet
  options={options}
  value={selectedValues}
  onValueChange={setSelectedValues}
  placeholder="Select fruits"
  title="Choose Fruits"
/>
```

| Prop | Type | Default | Description |

## Examples

### Country Selector

```tsx
const countries = [
  { label: 'United States', value: 'us' },
  { label: 'United Kingdom', value: 'uk' },
  { label: 'Canada', value: 'ca' },
  { label: 'Australia', value: 'au' },
  // ...more countries
];

export function CountrySelector() {
  const [country, setCountry] = useState('');

  return (
    <SelectSheet
      label="Country"
      options={countries}
      value={country}
      onValueChange={setCountry}
      placeholder="Select your country"
      title="Select Country"
      searchable
      searchPlaceholder="Search countries..."
    />
  );
}
```

### Category Filter

```tsx
export function CategoryFilter({ onFilter }) {
  const [category, setCategory] = useState('');

  const categories = [
    { label: 'All Categories', value: '' },
    { label: 'Electronics', value: 'electronics' },
    { label: 'Clothing', value: 'clothing' },
    { label: 'Home & Garden', value: 'home' },
    { label: 'Sports', value: 'sports' },
  ];

  const handleChange = (value) => {
    setCategory(value);
    onFilter(value);
  };

  return (
    <SelectSheet
      options={categories}
      value={category}
      onValueChange={handleChange}
      placeholder="Filter by category"
      size="sm"
    />
  );
}
```

### Async Options (Load More)

```tsx
export function AsyncSelect() {
  const [options, setOptions] = useState([]);
  const [loading, setLoading] = useState(false);
  const [hasMore, setHasMore] = useState(true);
  const [page, setPage] = useState(1);

  const loadMore = async () => {
    if (loading || !hasMore) return;
    setLoading(true);
    
    const newOptions = await fetchOptions(page);
    setOptions([...options, ...newOptions]);
    setPage(page + 1);
    setHasMore(newOptions.length > 0);
    setLoading(false);
  };

  return (
    <SelectSheet
      options={options}
      value={value}
      onValueChange={setValue}
      onLoadMore={loadMore}
      hasMore={hasMore}
      isLoading={loading}
      searchable
    />
  );
}
```
